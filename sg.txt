sg command                                                     function

sg_persist -n -o -G -K $host_key -S $node_key -d $dev          do_register
sg_persist -n -o -I -S $node_key -d $dev                       do_register_ignore
sg_persist -n -o -R -T 5 -K $host_key -d $dev                  do_reserve
sg_persist -n -o -L -T 5 -K $host_key -d $dev                  do_release
sg_persist -n -o -P -T 5 -K $host_key -S $node_key -d $dev     do_preempt
sg_persist -n -o -A -T 5 -K $host_key -S $node_key -d $dev     do_preempt_abort
sg_persist -n -i -k -d $dev                                    get_registration_keys
sg_persist -n -i -r -d $dev                                    get_reservation_key

-o --out
-i --in

-I --register-ignore

-T --prout-type=TYPE    5 = write exclusive registrants only

-L --release

-A --preempt-abort
